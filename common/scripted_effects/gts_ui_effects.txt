	# 领袖界面-点击已装备的道具

		set_variable = {
			which = gts_ui_current_slot
			value = $SLOT$
		}
		set_variable = {
			which = gts_ui_current_slot_class
			value = $CLASS$
		}
		set_variable = {
			which = gts_ui_current_item_page
			value = 0
		}

			id = gts_ui.9
		}
	}
}

gts_remove_leader_item = {
	# 道具界面-删除之前道具带来的特质
	hidden_effect = {
		if = {
			# 如果没有道具就跳过
			limit = {
				event_target:current_gts_leader = {
					check_variable = {
						which = gts_leader_slot_$SLOT$_item
						value = 0
					}
				}
			}
		}
		else_if = {
			# 移除宠物
			limit = {
				check_variable = {
					which = gts_ui_current_slot_class
					value = 1
				}
			}
			if = {
				limit = {
					event_target:current_gts_leader = {
						check_variable = {
							which = gts_leader_slot_$SLOT$_item
							value = @gts_leader_item_p_dragon
						}
					}
				}
				event_target:current_gts_leader = {
					remove_leader_flag = gts_leader_item_p_dragon
				}
			}
			else_if = {
				limit = {
					event_target:current_gts_leader = {
						check_variable = {
							which = gts_leader_slot_$SLOT$_item
							value = @gts_leader_item_p_fortress
