gts_ui_intel_panel_lock = {
	# 情报界面-防止打开多个领袖或者招募窗口
	hidden_trigger = {
		NOR = {
			has_country_flag = gts_pop_up_opened
			has_country_flag = gts_leader_panel_opened
		}
	}
}
gts_ui_leader_panel_lock = {
	# 领袖界面-防止打开多个道具或者身体窗口
	hidden_trigger = {
		NOR = {
			has_country_flag = gts_leader_body_panel_opened
			has_country_flag = gts_leader_item_panel_opened
		}
	}
}
gts_ui_intel_display = {
	# 情报界面-显示领袖
	from = {
		check_variable = {
			which = gts_ui_intel_page
			value = $PAGE$
		}
	}
}
gts_ui_leader_display_item = {
	# 领袖界面-展示的领袖获得的道具
	event_target:current_gts_leader = {
		check_variable = {
			which = gts_leader_slot_$SLOT$_item
			value = $ITEM$
		}
	}
}
gts_ui_item_display = {
	# 道具界面-显示道具
	from = {

			value = $PAGE$
		}

			which = gts_ui_current_slot_class

	}
}

gts_leader_body_add_pop_limit = {
	# 身体界面-点击殖民地十次限制
	hidden_trigger = {
		from = {
			if = {
				limit = {
					check_variable = {
						which = gts_ui_current_body
						value = @gts_body_armpit
					}
				}
				event_target:current_gts_leader = {
					check_variable = {
						which = gts_leader_body_type_1_$CLASS$
						value < 10
					}
				}
			}else_if = {
				limit = {
					check_variable = {
						which = gts_ui_current_body
						value = @gts_body_breast
					}
				}
				event_target:current_gts_leader = {
					check_variable = {
						which = gts_leader_body_type_2_$CLASS$
						value < 10
					}
				}
			}else_if = {
				limit = {
					check_variable = {
						which = gts_ui_current_body
						value = @gts_body_digestion
					}
				}
				event_target:current_gts_leader = {
					check_variable = {
						which = gts_leader_body_type_3_$CLASS$
						value < 10
					}
				}
			}else_if = {
				limit = {
					check_variable = {
						which = gts_ui_current_body
						value = @gts_body_reporduction
					}
				}
				event_target:current_gts_leader = {
					check_variable = {
						which = gts_leader_body_type_4_$CLASS$
						value < 10
					}
				}
			}else_if = {
				limit = {
					check_variable = {
						which = gts_ui_current_body
						value = @gts_body_feet
					}
				}
				event_target:current_gts_leader = {
					check_variable = {
						which = gts_leader_body_type_5_$CLASS$
						value < 10
					}
				}
			}
		}
	}
}
