	leader_class = { scientist }
	is_leader_possible = {
		OR = {
			leader_class = scientist
		}
	}

	civic = civic_panlon_Wizard
	modifier = {
		ship_anomaly_generation_chance_mult = 0.03
		ship_anomaly_research_speed_mult = 0.03
		science_ship_survey_speed = 0.03
	}
	icon = "GFX_icon_councilor_research"
}
