gts_bruh_destroyer_ship = {
	
	entity = "GTS_void_entity"
	
	graphical_culture = no
	formation_priority = @destroyer_formation_priority
	max_speed = 110
	acceleration = 0.3
	rotation_speed = 0.2
	collision_radius = @destroyer_collision_radius
	max_hitpoints = 1000000
	
	modifier = {
		
	}
	
	size_multiplier = 32
	fleet_slot_size = 32
	
	section_slots = {
		"bow" = {
			locator = "part1"
		}
		"stern" = {
			locator = "part2"
		}
	}
	
	num_target_locators = 4
	icon = GFX_tech_gts_destroyer_ship
	is_space_station = no
	base_buildtime = 5760
	can_have_federation_design = no
	enable_default_design = yes	#if yes, countries will have an auto-generated design at start
	enable_3dview_in_ship_browser = yes #if yes, ship design will have a 3d view in the ship browser
	
	default_behavior = picket

	role_background = "GFX_role_selection_destroyer"
	ship_roles = { screen gunship artillery brawler }
	
	prerequisites = {
		"tech_gts_bruh_destroyer_ship"
	}
	
	combat_disengage_chance = 1.50
	
	class = shipclass_military
	construction_type = starbase_shipyard
	required_component_set = "power_core"
	required_component_set = "ftl_components"
	required_component_set = "thruster_components"
	required_component_set = "sensor_components"
	required_component_set = "combat_computers"
	required_component_set = "ship_aura_components"
	
	potential_construction = {
		OR = {
			is_scope_type = starbase
			AND = {
				is_scope_type = megastructure
				OR = {
					is_megastructure_type = mega_shipyard_3
					is_megastructure_type = mega_shipyard_restored
				}
			}
		}
	}

	possible_construction = {
		custom_tooltip = {
			fail_text = starbase_citadel_trigger
			OR = {
				AND = {
					is_scope_type = megastructure
					OR = {
						is_megastructure_type = mega_shipyard_3
						is_megastructure_type = mega_shipyard_restored
					}
				}
				AND = {
					is_scope_type = starbase
					has_starbase_size >= starbase_citadel
				}
			}
		}
		custom_tooltip = {
			fail_text = starbase_titan_yards_trigger
			OR = {
				AND = {
					is_scope_type = megastructure
					OR = {
						is_megastructure_type = mega_shipyard_3
						is_megastructure_type = mega_shipyard_restored
					}
				}
				AND = {
					is_scope_type = starbase
					has_starbase_building = titan_yards
				}
			}
		}
	}
	
	resources = {
		category = ships
		
		upkeep = {
			alloys = 1000
			consumer_goods = 1000
			gts_micro = 10
			gts_giantess_underwear = 30
		}
	}
	
	min_upgrade_cost = {
		alloys = 50
	}
	
	potential_country = {
		is_ai = no
	}
}
