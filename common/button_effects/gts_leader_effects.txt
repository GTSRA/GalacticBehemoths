gts_leader_colossus_effect = {
	potential = {
		always = yes
	}
	allow = {
		#星际行动限制
		custom_tooltip = {
			fail_text = "gts_leader_interact_no_intro"
			has_global_flag = gts_leader_interact_intro
		}
		custom_tooltip = {
			fail_text = "gts_gateway_final_needed"
			from = {
				has_megastructure = gts_gateway_final
				has_country_flag = gts_gateway_situation_done
			}
		}
		custom_tooltip = {
			fail_text = "gts_leader_interact_cooldown"
			from = {
				NOR = {
					has_country_flag = gts_leader_interact_cooldown
					has_country_flag = gts_leader_interact_active
					has_country_flag = gts_leader_interact_storm_coming
				}
			}
		}
		gts_ui_leader_panel_lock = yes
	}
	effect = {
		custom_tooltip = gts_leader_colossus.tooltip
		country_event = {
			id = gts_ui.10
		}
	}
}

				limit = {
