gts_leader_body_btn_goddess_effect = {
	potential = {
		from = {
			has_country_flag = gts_flag
		}
	}
	allow = {
		resource_stockpile_compare = {
			resource = gts_micro
			value >= gts_bruh_get_gts_micro_cost_amount_variable mult = -1
		}
		gts_leader_body_add_pop_limit = { CLASS = 1 }
	}
	effect = {
		custom_tooltip = gts_leader_body_goddess_btn.tooltip
		add_resource = {
			gts_micro = 1
			mult = gts_bruh_get_gts_micro_cost_amount_variable
		}
		gts_leader_body_add_pop = { CLASS = 1 }
	}
}
