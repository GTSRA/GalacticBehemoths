gts_leader_colossus_mode_planet_01_effect = {
	allow = {
		custom_tooltip = {
			fail_text = "gts_leader_interact_storm_coming"
			from = {
				NOR = {
					has_country_flag = gts_leader_interact_storm_coming
				}
			}
		}
		custom_tooltip = {
			fail_text = "gts_leader_interact_cooldown"
			from = {
				NOR = {
					has_country_flag = gts_leader_interact_cooldown
				}
			}
		}
		custom_tooltip = {
			fail_text = "gts_leader_interact_no_at_war"
			owner = {
				is_at_war_with = from
			}
		}
	}
	potential = {
		is_scope_type = planet
	}
	effect = {
		custom_tooltip = gts_ui_mode_planet_01_effect.tooltip
		hidden_effect = {
			save_event_target_as = gts_leader_planet
			solar_system = {
				save_event_target_as = gts_leader_system
				create_gts_space_storm_object_effect = yes
			}
			from = {
				save_event_target_as = gts_leader_country
				set_timed_country_flag = {
					flag = gts_leader_interact_storm_coming
					days = 30
				}
				country_event = {
					id = gts_leader_interact.1001
					days = 30
				}
			}			
		}
	}
}

			fail_text = "gts_leader_interact_storm_coming"
			from = {
				NOR = {
					has_country_flag = gts_leader_interact_storm_coming
				}
			}
		}
		custom_tooltip = {
			fail_text = "gts_leader_interact_cooldown"
			from = {
				NOR = {
					has_country_flag = gts_leader_interact_cooldown
				}
			}
		}
		custom_tooltip = {
			fail_text = "gts_leader_interact_no_ultra"
			event_target:current_gts_leader = {
				has_trait = leader_trait_Ultra
			}
		}
		custom_tooltip = {
			fail_text = "gts_leader_interact_friendly"
			solar_system = {
				owner = {
					NOR = {
						is_same_value = from
						is_friendly_to = from
					}
				}
			}
		}
	}
	potential = {
		is_scope_type = planet
	}
	effect = {
		custom_tooltip = gts_ui_mode_system_01_effect.tooltip
		hidden_effect = {
			save_event_target_as = gts_leader_planet
			solar_system = {
				save_event_target_as = gts_leader_system
				create_gts_space_storm_object_effect = yes
			}
			if = {
				limit = {
					event_target:current_gts_leader = {
						has_leader_flag = gts_qhai
					}
					from = {
						any_owned_leader = {
							has_leader_flag = gts_huatian
						}
					}

				}
				from = {
					save_event_target_as = gts_leader_country
					set_timed_country_flag = {
						flag = gts_leader_interact_storm_coming
						days = 30
					}
					country_event = {
						id = gts_leader_qhai.551
						days = 30
					}
				}
			}
			else = {
				from = {
					save_event_target_as = gts_leader_country
					set_timed_country_flag = {
						flag = gts_leader_interact_storm_coming
						days = 30
					}
					country_event = {
						id = gts_leader_interact.2001
						days = 30
					}
				}
			}
		}	
	}

					OR = {
						NOR = { exists = leader }
						AND = {
							exists = leader
							leader = {  NOR = { has_trait = leader_trait_Ultra } }
						}
					}
				}
