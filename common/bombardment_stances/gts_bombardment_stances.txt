tinyraiding = {
	trigger = {
		OR = {
			owner = { has_civic = civic_delta_trade }
		}
	}

	default = no

	stop_when_armies_dead = no
	abduct_pops = yes

	planet_damage = 0.5
	army_damage = 1.5

	kill_pop_chance = {
		base = 1
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 8
		modifier = {
			factor = 0
			exists = from
			from = {
				OR = {
					pop_amount < 200
					owner = { NOT = { is_hostile = root.owner } }
				}
			}
			NOT = {
				owner = { has_civic = civic_delta_trade }
			}
		}
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { has_claim = root.solar_system }
			}
			NOT = {
				owner = { has_civic = civic_delta_trade }
			}
		}
		modifier = {
			factor = 100
			owner = { has_civic = civic_delta_trade }
		}
	}
}

gts_115tiny_armageddon = {
	trigger = {
		OR = {
			owner = {
				OR = {
					has_crisis_perk = gts_115tiny_menp_death_from_above		# Menace Perk: Death From Above
				}
			}
			AND = {
				owner = { is_country_type = enclave_mercenary }
				controller = {
					OR = {
						has_menace_perk = gts_115tiny_menp_death_from_above		# Menace Perk: Death From Above
					}
				}
			}
		}
	}

	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 7.0
	army_damage = 7.0

	kill_pop_chance = {
		base = 4.5
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 20
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
				NAND = {
					controller = { is_crisis_faction = yes }
					planet_garrison_strength > 0
				}
			}
			exists = owner
			owner = { is_crisis_faction = no }
		}
	}
}
