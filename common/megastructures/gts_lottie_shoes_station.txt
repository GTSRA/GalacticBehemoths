					from = {

					}

			orbit_location = yes
			orbit_angle_offset = 45
			orbit_distance_offset = 5
			size = 10
			has_ring = no

					entity = "shoesspaceport_entity_01_entity"

					surveyed = yes

				limit = {
					has_orbital_station = yes
				}
				orbital_station = {
					dismantle = yes
				}
			}
		}
		remove_megastructure = fromfrom
		#from = { country_event = { id = utopia.61 } }
	}
