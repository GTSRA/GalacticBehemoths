gts_decision_egg_nutrition = {

	# 給卵子星球提供營養

	owned_planets_only = yes
	sound = event_activating_unknown_technology
	icon = decision_gts_egg_nutrition

	enactment_time = -1

	resources = {
		category = decisions
		cost = {
			influence = 100
		}
	}

	potential = {
		NOT = {is_no_eat_planet = yes}
		owner = {
			any_owned_planet = {
				has_planet_flag = Luanzi_home_planet
			}
		}
	}

	allow = {}

	effect = {
		custom_tooltip = gts_decision_egg_nutrition_effects
		hidden_effect = {
			save_event_target_as = gts_nutrition_planet
			export_trigger_value_to_variable = {
				trigger = pop_amount
				variable = gts_pop_nutrition_amount
			}
			export_trigger_value_to_variable = {
				trigger = planet_size
				variable = gts_planet_size_nutrition_amount
			}
			# 10000人口 = 1 size
			
			divide_variable = {
				which = gts_pop_nutrition_amount
				value = 10000
			}
			round_variable = gts_pop_nutrition_amount

			# 5星球尺寸 = 1 size

			divide_variable = {
				which = gts_planet_size_nutrition_amount
				value = 5
			}
			round_variable = gts_planet_size_nutrition_amount

			owner = {
				random_owned_planet = {
					limit = {
						has_planet_flag = Luanzi_home_planet
					}
					while = {
						count = event_target:gts_nutrition_planet.gts_pop_nutrition_amount
						add_deposit = d_gts_nutrients
						change_planet_size = 1
					}
					while = {
						count = event_target:gts_nutrition_planet.gts_planet_size_nutrition_amount
						add_deposit = d_gts_nutrients
						change_planet_size = 1
					}
				}
				country_event = {
					id = gts_egg.10
				}
			}
			if = {
				limit = {
					exists = orbital_defence
				}
				destroy_fleet = orbital_defence
			} 
			else = {
				random_fleet_in_orbit = {
					limit = {
						is_owned_by = root.owner
						is_ship_class = shipclass_starbase
					}
					destroy_fleet = this
				}
			}
		}
	}

	ai_weight = {
		weight = 2
		modifier = {
			add = 3
			owner = {
				has_origin = origin_Luanzi
			}
		}
	}
}
