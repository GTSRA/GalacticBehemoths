counter_gts_pop = {
	# 計算巨娘人口(有上限版)
	complex_trigger_modifier = {
		trigger = count_owned_pop_amount
		parameters = {
			limit = {
				is_gts = yes
				pop_has_happiness = yes
			}
		}
		mode = add
		mult = 0.01
	}
	max = 100
}
