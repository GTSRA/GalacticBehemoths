gts_r_plasmic_core = {
    activation_duration = @triumph_duration
    portrait = "GFX_plasmic_core"
    sound = "relic_activation_plasmic_core"

    resources = {
        category = relics
        # Activation cost
        cost = {
            unity = @activation_cost
        }
    }

    ai_weight = {
        weight = 100
    }

    triggered_country_modifier = {
        potential = {
            is_robot_empire = no
        }
        logistic_growth_mult = 0.1
		bonus_pop_growth_mult = 0.1
    }
    triggered_country_modifier = {
        potential = {
            is_robot_empire = yes
        }
        species_leader_exp_gain = 0.1
    }

    score = 3000

    active_effect = {
        if = {
            limit = { is_robot_empire = yes }
            custom_tooltip = machine_plasmic_core_effect
            hidden_effect = {
                random_owned_leader = {
                    add_trait = {
						trait = random_common
						show_message = no
					}
                    add_trait = {
						trait = random_common
						show_message = no
					}
                    add_trait = {
						trait = random_common
						show_message = no
					}
                }
            }
        }
        else = {
            custom_tooltip = gts_plasmic_core_effect
            hidden_effect = {
                random_owned_planet = {
                    limit = {
                        any_owned_pop_group = {
                            is_robot_pop = no
                            NOT = { pop_has_trait = gts_trait_plasmic }
                        }
                    }
                    every_owned_pop_group = {
                        limit = {
                            is_robot_pop = no
                        }
                        modify_species = {
                            species = this
                            add_trait = trait_plasmic
                            portrait = this
                        }
                    }
                }
            }
        }
        custom_tooltip = relic_triumph_cooldown
        hidden_effect = {
            add_modifier = {
                modifier = "relic_activation_cooldown"
                days = @triumph_duration
            }
        }
    }

    # Possible check for activation
    possible = {
        custom_tooltip = {
            fail_text = "gts_plasmic_core_no_suitable_planet"
            OR = {
                is_robot_empire = yes
                any_owned_planet = {
                    any_owned_pop_group = {
                        is_robot_pop = no
                        NOT = { pop_has_trait = trait_plasmic }
                    }
                }
            }
        }
        inline_script = {
            script = relics/activation_checks
            RELIC = gts_r_plasmic_core
        }
    }
}
