			has_country_flag = vending_machine_contents_full
			NOR = {

				is_megacorp = yes
			}
		}
		text = anomaly.7010.desc.full.bio
	}
	desc = {
		trigger = {
			has_country_flag = vending_machine_contents_empty
		}
		text = anomaly.7010.desc.empty
	}
	desc = {
		trigger = {
			has_country_flag = vending_machine_contents_rot
			OR = {
				is_robot_empire = yes
				is_lithoid_empire = yes
			}
		}
		text = anomaly.7010.desc.rot.nonbio
	}
	desc = {
		trigger = {
			has_country_flag = vending_machine_contents_rot
			NOR = {
				is_robot_empire = yes
				is_lithoid_empire = yes
			}
		}
		text = anomaly.7010.desc.rot.bio
	}
	picture = GFX_evt_enigmatic_cache
	show_sound = event_activating_unknown_technology
	location = FROM
	is_triggered_only = yes
	immediate = {
		random_list = {
			10 = {
				set_country_flag = vending_machine_contents_full
			}
			60 = {
				set_country_flag = vending_machine_contents_empty
			}
			30 = {
				set_country_flag = vending_machine_contents_rot
			}
		}
	}
	after = {
		hidden_effect = {
			remove_country_flag = vending_machine_contents_full
			remove_country_flag = vending_machine_contents_empty
			remove_country_flag = vending_machine_contents_rot
		}
	}
	option = {
		name = anomaly.7010.a
		trigger = {
			has_country_flag = vending_machine_contents_full
			is_megacorp = yes
		}
		add_monthly_resource_mult = {
			resource = energy
			value = @tier2materialreward
			min = @tier2materialmin
			max = @tier2materialmax
		}
	}
	option = {
		name = anomaly.7010.b
		switch = {
			trigger = has_country_flag
			vending_machine_contents_full = {
				if = {
					limit = {
						NOR = {
							is_robot_empire = yes
							is_lithoid_empire = yes
						}
					}
					add_modifier = {
						modifier = taste_the_stars
						days = 7200
					}
				}
				else = {
					add_resource = {
						exotic_gases = 100
					}
				}
			}
			vending_machine_contents_empty = {
				if = {
					limit = {
						has_ancrel = yes
					}
					custom_tooltip = minor_artifacts_found
					add_resource = {
						minor_artifacts = 1
					}
				}
				else = {
					add_resource = {
						alloys = 200
					}
				}
			}
			vending_machine_contents_rot = {
				give_next_tech_society_option_effect = yes
			}
		}
	}
}
