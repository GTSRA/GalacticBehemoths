					max_orientation_angle = 40

					hyperlane = no
					effect = {
						save_global_event_target_as = ucluster2b
					}
				}
			}
			event_target:ucluster2 = {
				spawn_system = {

			}

				spawn_system = {
					min_distance >= 20
					max_distance <= 30

					max_orientation_angle = 66

				}

		event_target:ucluster4 = {
			add_hyperlane = {
				from = event_target:ucluster4
				to = event_target:ucluster3
			}
			add_hyperlane = {
				from = event_target:ucluster4
				to = event_target:ucluster2
			}
			add_hyperlane = {
				from = event_target:ucluster4
				to = event_target:ucluster4b
			}
			add_hyperlane = {
				from = event_target:ucluster4
				to = event_target:ucluster4c
			}
		}
		event_target:ucluster3 = {

				from = event_target:ucluster1b

			}
		}
		set_spawn_system_batch = end
		# 迁都到目标星球
		event_target:ucluster1 = {
			create_starbase = {
				owner = event_target:gts_ultra_country
				size = starbase_starport
				module = shipyard
				module = anchorage
				building = crew_quarters
			}
		}
		random_planet_within_border = {
			limit = {
				has_planet_flag = ultra_home_planet
			}
			create_colony = {
				owner = event_target:gts_ultra_country
			}
			while = {
				count = 5
				create_pop_group = {
					species = owner
				}
			}
			set_capital = yes
		}
		#移除原有星球
		every_owned_planet = {
			limit = {
				NOT = {
					has_planet_flag = ultra_home_planet
				}
			}

			clear_planet_modifiers = yes
			clear_deposits = yes
		}

		# every_galaxy_planet = {
		# 	set_surveyed = { surveyed = no surveyor = event_target:gts_ultra_country }

			id = gts_ultra_road.10
			days = 7
		}
