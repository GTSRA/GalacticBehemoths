country_event = {
	id = gts_rawls.107
	title = "TRANSMISSION"	#信号传入接入画面
	desc = gts_rawls.107.desc
	diplomatic = yes
	is_triggered_only = yes

	picture_event_data = {
		portrait = gtsra_leader_lottie_portrait_02

	}

	immediate = {
		# 用flag找到洛蒂
		every_owned_leader = {
			limit = {
				has_leader_flag = gts_luodi
			}
			save_event_target_as = gts_bruh_lottie_kill_count
		}
		
		# 你不一定會選A
		# 但這裡得先計算好
		random_owned_planet = {
			save_event_target_as = lottie_target_planet
		}
		
		event_target:lottie_target_planet = {
			# 計算人口數量 (90%)
			export_trigger_value_to_variable = {
				trigger = pop_amount
				variable = gts_bruh_get_planet_killed_pop_count
			}
			multiply_variable = {
				which = gts_bruh_get_planet_killed_pop_count
				value = 0.9
			}
			round_variable = gts_bruh_get_planet_killed_pop_count
			
			# 計算經驗獲取
			export_trigger_value_to_variable = {
				trigger = pop_amount
				variable = gts_bruh_get_experience_count
			}
			multiply_variable = {
				which = gts_bruh_get_experience_count
				value = 0.9
			}
			divide_variable = {
				which = gts_bruh_get_experience_count
				value = 10
			}
			round_variable = gts_bruh_get_experience_count
		}
	}
	option = {
		name = gts_rawls.107.A
		custom_tooltip = gts_rawls.107.A.tooltip
		event_target:lottie_target_planet = {
			every_owned_pop_group = {

		}

			add_experience = event_target:lottie_target_planet.gts_bruh_get_experience_count

				which = gts_bread_logs
				value = event_target:lottie_target_planet.gts_bruh_get_planet_killed_pop_count
			}
		}
		hidden_effect = {
			country_event = {
				id = gts_rawls.100

	}

		custom_tooltip = gts_rawls.107.B.tooltip
