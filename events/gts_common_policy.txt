namespace = gts_policy
country_event = {
	id = gts_policy.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_policy_flag = gts_gaintess_apartheid
	}
	immediate = {
		owner = {
			every_owned_pop_group = {
				if = {
					limit = {
						is_gts = yes
					}
					add_modifier = {
						modifier = gts_be_apartheid
						days = 30
					}
				}
				if = {
					limit = {
						is_tiny = yes
					}
					add_modifier = {
						modifier = gts_apartheid_a
						days = 30
					}
				}
			}
		}
	}
}
