country_event = {
	id = gts_aya.10	#ID
	hide_window = yes	#隐藏界面
	is_triggered_only = yes	#只能触发
	trigger = {
		#触发器：有起源
		has_origin = origin_Aya
	}
	immediate = {
		#很快啊
		random_owned_planet = {
			#随机拥有的星球
			limit = {
				has_planet_flag = Aya_home_planet
			}			#限制：需为母星
			Aya_origin_effect = yes			#提供起源buff
		}
		if = {
			limit = {
				owner = {
					has_origin = origin_Aya
				}
			}
			owner = {
				capital_scope = {
					remove_all_districts = yes
					while = {
						count = 2
						add_district = district_arcology_housing
					}
				}
			}
		}
	}
}

	}

	}

		}
		hidden_effect = {
			create_leader = {
				class = official				#leader
				name = NAME_AYA_Maruyama
				event_leader = yes
				
				gender = female
				skill = 6
				set_age = 17
				traits = {
					trait = trait_Maruyama_Aya
					trait = leader_trait_chosen
					trait = leader_trait_Giant
				}
			}
			every_owned_leader = {
				limit = {
					has_trait = trait_Maruyama_Aya
				}
				remove_trait = leader_trait_Small
				add_trait = {
			trait = leader_trait_Giant
			show_message = no
		}
			}
			last_created_leader = {
				save_global_event_target_as = aya_governor
				set_leader_flag = has_gained_level_trait
				set_leader_flag = gts_aya
				gts_ui_leader_init = yes
				change_leader_portrait = gts_AyaMaruyama_Live
			}
			set_leader = last_created_leader
		}
	}
}
