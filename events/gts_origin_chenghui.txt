									exists = fleet

										num_ships > 0
									}
								}
							}
						}
					}
				}
				AND = {
					has_country_flag = chenghui_warship_active					#是成惠舰队
					#NOT = { exists = event_target:chenghui_warship } #舰队无了

							is_same_value = event_target:chenghui_warship

							num_ships > 0
						}
					}
				}
				AND = {
					has_country_flag = chenghui_governor_active					#是成惠总督
					#NOT = { exists = event_target:chenghui_governor } #总督无了
				}
			}
		}
	}
	immediate = {
		event_target:chenghui_owner = {
			remove_country_flag = chenghui_governor_active

			country_event = {
