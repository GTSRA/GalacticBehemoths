            spawn_system = {

                initializer = gts_cys_system
            }
        }
        set_country_flag = gts_cys_owner
    }
}

    id = gts_cys.2
    title = "gts_cys.2.name"
    desc = "gts_cys.2.desc"
    picture = GFX_evt_ruined_system
    show_sound = event_radio_chatter

    is_triggered_only = yes

    trigger = {
        has_global_flag = gts_has_trigger_cys
        from = { has_star_flag = gts_cys_system }
        fromfrom = {
            has_country_flag = gts_cys_owner
        }
    }
    location = event_target:gts_cys_star

    option = {
        name = gts_cys.2.a
        owner = {
            add_modifier = {
                modifier = gts_cys_buff1
                days = -1
            }
        }
    }
    option = {
        name = gts_cys.2.b
    }
    option = {
        name = gts_cys.2.c
        owner = {
            start_situation = {
                type = situation_cys
                target = this
            }
        }
    }
}
