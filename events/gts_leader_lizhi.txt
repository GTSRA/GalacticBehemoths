# 替領袖添加不屈特質

country_event = {
	id = gts_lizhi.0000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		from = {
			NOT = {
				has_trait = leader_trait_tinyunbending
			}
		}
		from.species = {
			has_trait = trait_tinyunbending
		}
	}

	immediate = {
		if = {
			limit = {
				from = {
					leader_class = commander
					NOT = {
						has_trait = leader_trait_tinyunbending
					}
				}
				from.species = {
					has_trait = trait_tinyunbending
				}
			}
			from = {
				add_trait = {
					trait = leader_trait_tinyunbending
					show_message = no
				}
			}
		}
	}
}
